# source TESTS/Tx_TESTS/MODELSIM_BACKUP/FUNCTIONAL_TEST_CASES/TEST_40_64B/TEST_44_44_44B/script_5G.txt  
# script for testing LMAC_STREAM :  generating tx packets
# gen_en_wr testbench signals generates tx packets which is send from AXIS bridge to LMAC



vsim work.axis_lmac_tb 
view wave
do C:/LMAC2_INFO/waveforms/testwave_5G 
view -new wave
#add wave -r /*

mem load -i C:/LMAC2_INFO/TESTS/Tx_TESTS/MODELSIM_BACKUP/FUNCTIONAL_TEST_CASES/TEST_40_64B/TEST_44_44_44B/memory_wr_data44_44_44B.txt -format hex /AXIS_LMAC_TB/axi_stream_master/memory_wr_module/memory_wr_data
mem load -i C:/LMAC2_INFO/TESTS/Tx_TESTS/MODELSIM_BACKUP/FUNCTIONAL_TEST_CASES/TEST_40_64B/TEST_44_44_44B/memory_wr_ctrl44_44_44B.txt -format hex /AXIS_LMAC_TB/axi_stream_master/memory_wr_module/memory_wr_ctrl
                   
mem load -i C:/LMAC2_INFO/PHY_EMULATOR/rx_pkt_gen_data.mem -format mti /AXIS_LMAC_TB/phy_emulator_10G/rx_pkt_gen_10G/rx_pkt_gen2kx64_inst/dram_data/mem
mem load -i C:/LMAC2_INFO/PHY_EMULATOR/rx_pkt_gen_ctrl.mem -format mti /AXIS_LMAC_TB/phy_emulator_10G/rx_pkt_gen_10G/rx_pkt_gen2kx8_inst/dram_ctrl/mem


force -freeze sim:/AXIS_LMAC_TB/fmac_speed 3'b101 0
force -freeze sim:/AXIS_LMAC_TB/phy_emulator_10G/rx_pkt_gen_start_addr 64'd00 0


# generating first packet     - TX packet 1

run 200ns
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0
run 4ns
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0

# Generate the TX packet	-- TX packet 2
run 200ns
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0
run 4ns
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0

# Generate the TX packet	-- TX packet 3
run 200ns
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 1 0
run 4ns
force -freeze sim:/AXIS_LMAC_TB/gen_en_wr 0 0

run 1us
